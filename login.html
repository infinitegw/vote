// LOGIN HANDLER document.getElementById("loginForm")?.addEventListener("submit", function (e) { e.preventDefault(); const admissionNo = document.getElementById("admissionNo").value.trim(); const fullName = document.getElementById("fullName").value.trim(); const className = document.getElementById("className").value; const dorm = document.getElementById("dorm").value; let students = JSON.parse(localStorage.getItem("students")) || []; // Find matching student const student = students.find( (s) => s.admissionNo === admissionNo && s.fullName.toLowerCase() === fullName.toLowerCase() && s.className === className && s.dorm === dorm ); if (student) { // Save logged-in user localStorage.setItem("currentUser", JSON.stringify(student)); alert("Login successful!"); window.location.href = "dashboard.html"; // redirect to dashboard } else { alert("Invalid details! Please check your Admission No, Name, Class, and Dorm."); } });
